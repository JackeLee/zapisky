\chapter{Stone duality}
\section{Stone correspondence for \StoneFrm}
\begin{definition}
    We say a frame is \DEF{Stone frame} if it is compact, null--dimensional and regular.
\end{definition}

\begin{definition}
    Let $B$ be a Boolean algebra. Define $\DEFSYM{J}{\J} B$ to be the set of all ideals of $B$.
    For $f\colon A \to B$ a Boolean homomorphism, define
    $$(\J f)(I) = \downset f[I].$$
\end{definition}

\begin{proposition}
    $\J\colon \Bool \to \StoneFrm$ is a functor.
\end{proposition}
\begin{proof}
    $\J B$ is trivially compact frame. Since $\downset a \rbelow \downset a$ and $\downset a$ is complemented for each $a$ of $B$, $\J B$ is regular and null--dimensional.

    The rest is easy.
\end{proof}

\begin{definition}
    Let $L$ be a Stone frame. Define $\DEFSYM{Bc}{\Bc} L$ to be the set of all complemented elements of $L$ and for $f\colon L \to M$ a frame homomorphism define
    $$\Bc f = \restr{f}{\Bc L}\colon \Bc L \to \Bc M.$$
\end{definition}
% TODO somehow mention that $f$ is between stone spaces

From the fact that a homomorphic image of a complemented element is a complemented element one can see that $\Bc f$ is well--defined. % TODO some comment's for next statement

\begin{observation}
    $\Bc\colon \StoneFrm \to \Bool$ is a functor.
\end{observation}

\begin{lemma}
    Let $B$ be a Boolean algebra and $I \in \J B$. Then $I$ is complemented iff $I = \downset b$ for some $b \in B$.
\end{lemma}
\begin{proof}
    Let $I$ be a complemented ideal. Since $I \vee I^c = 1_{\J B}$ there exists $a \in I$ and $b \in I^c$ such that $a \vee b = 1$. From $I \wedge I^c = \{0\}$ we have $a \wedge b = 0$ and $I \wedge \downset b = \{0\}$.
    From uniques of complements we get $I^c = \downset b$, indeed $I \vee \downset b = 1_{\J B}$ and $I \wedge \downset b = \{0\}$. Using the same arguments we get $I = \downset a$.

    Converse implication is trivial since $\downset a \vee \downset a^c = 1_{\J B}$ and $\downset a \wedge \downset a^c = \{0\}$.
\end{proof}

\num For $B \in \Bool$ define \index{istar@$i_*$}$i_B\colon B \to \Bc\J(B)$ as follows
    $$i_B\colon b \mapsto \downset b.$$
    The definition is correct by previous lemma and $i_B$ is a Boolean homomorphism, indeed $\downset a \vee \downset b = \downset (a \vee b)$, $\downset a \wedge \downset b = \downset (a \wedge b)$ and $\downset 1 = B$ respectively $\downset 0 = \{0\}$ is top respectively bottom of $\Bc\J(B)$.

From previous lemma we also see that $i_B$ is isomorphism of $B$ and $\Bc\J(B)$. And therefore the following diagram commutes

\begin{center}
\begin{tikzcd}
    A \ar{r}{i_A} \ar{d}{f} & \Bc\J(A) \ar{d}{\Bc\J(f)}\\
    B \ar{r}{i_B}           & \Bc\J(B)
\end{tikzcd}
\end{center}

\noindent for any $f\colon A \to B$ Boolean homomorphism. The collection $i_*$ of Boolean homomorphisms forms a natural equivalence between $\Bc\J$ and the identity functor on \Bool.

\num Similarly for $L \in \StoneFrm$ we have a mapping \index{vstar@$v_*$}$v_L\colon \J\Bc(L) \to L$ defined as
    $$v_L\colon I \mapsto \bigvee I.$$
    And a mapping in the opposite direction $\iota\colon L \to \J\Bc(L)$:
    $$\iota\colon e \mapsto \downset e \cap \Bc(L).$$

We can see that both $v_L$ and $\iota$ are monotone maps, $v_L \iota = \id_L$ (by null--dimensionality and regularity of $L$) and $\id_{\J\Bc(L)} \subseteq \iota v_L$, and therefore $v_L$ is the left Galois adjoint to $\iota$ and preserves all suprema.
Since $\bigvee I_1 \wedge \bigvee I_2 = \bigvee \{a_1 \wedge a_2 : a_i \in I_i\} \leq \bigvee \{ a : a \in I_1 \cap I_2 \} = \bigvee (I_1 \cap I_2) \leq \bigvee I_1 \cap \bigvee I_2$, $v_L$ also preserves finite infima which makes it a frame homomorphism.

Finally $\id_{\J\Bc(L)} = \iota v_L$: Take any $x \in \iota v_L(I)$. From the definitions we immediately see that $x \leq \bigvee I$ and $x$ is complemented in $L$. By the fact that
    $$1 = x \vee x^c \leq \bigvee I \vee x^c$$
    and by compactness of $L$ there is a finite $F \subseteq I$ such that $\bigvee F \vee x^c = 1$. Since $x = 1 \wedge x = (\bigvee F \vee x^c) \wedge x = (\bigvee F \wedge x) \vee (x^c \wedge x) = \bigvee F \wedge x$ we get that $x \leq \bigvee F$ and therefore $x \in I$.

Previous observations give us the fact that $v_L$ is isomorphism of $L$ and $\J\Bc(L)$ and also that for any $f\colon L \to M$ homomorphism of Stone frames the following diagram commutes

\begin{center}
\begin{tikzcd}
    \J\Bc(L) \ar{d}{\J\Bc(f)} \ar{r}{v_L} & L \ar{d}{f} \\
    \J\Bc(M) \ar{r}{v_M}                  & M
\end{tikzcd}
\end{center}

\noindent Again the collection $v_*$ of frame homomorphisms is a natural equivalence between $\J\Bc$ and identity functor on \StoneFrm.

\num From previous facts we get the main theorem.

\begin{blockProp*}{Theorem}
    Functor \Bc{} is left/right adjoint to the functor \J{} with units of adjunction $v_*$ and $i_*$. Moreover \Bc{} and \J{} constitute an isomorphism of categories \StoneFrm{} and \Bool.
\end{blockProp*}
